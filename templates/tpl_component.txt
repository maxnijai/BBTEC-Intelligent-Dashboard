// Leaflet Map (OpenStreetMap)
const LMap=({coords,districts,onDotClick})=>{
  const mapRef=useRef(null);
  const mapInst=useRef(null);
  const markersRef=useRef([]);
  

  useEffect(()=>{
    if(!mapRef.current||mapInst.current) return;
    // Load Leaflet CSS
    if(!document.getElementById("leaflet-css")){
      const link=document.createElement("link");
      link.id="leaflet-css";link.rel="stylesheet";
      link.href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css";
      document.head.appendChild(link);
    }
    // Load Leaflet JS
    const loadLeaflet=()=>{
      if(window.L) return initMap();
      const sc=document.createElement("script");
      sc.src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js";
      sc.onload=initMap;
      document.head.appendChild(sc);
    };
    const initMap=()=>{
      const L=window.L;
      const map=L.map(mapRef.current,{zoomControl:true}).setView([18.0,99.5],7);
      L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png",{
        attribution:'&copy; OpenStreetMap',maxZoom:18,opacity:0.9
      }).addTo(map);
      mapInst.current=map;
      
      // Ticket dots
      (coords||[]).forEach(c=>{
        const color=afC(c.af);
        const marker=L.circleMarker([c.lat,c.lon],{radius:6,fillColor:color,color:"white",weight:1,fillOpacity:0.85}).addTo(map);
        marker.bindPopup(`<div style="font-family:Inter,sans-serif;font-size:12px;line-height:1.6"><b>${c.sowner}</b><br/>${c.sev} | ${c.district||c.prov||"N/A"}<br/>Aging: <b>${c.aging}d</b><br/>Problem: ${c.problem}<br/>Root: ${c.root}${c.subimpact>0?"<br/>Subimpact: "+c.subimpact:""}</div>`,{maxWidth:250});
        marker.on("click",()=>{if(onDotClick)onDotClick(c);});
        markersRef.current.push(marker);
      });
      
      // District labels
      (districts||[]).filter(d=>d.lat&&d.lon&&d.name!=="ไม่ระบุ").forEach(d=>{
        L.circleMarker([d.lat,d.lon],{radius:Math.max(4,Math.min(14,d.count/3)),fillColor:"transparent",color:"white",weight:1,dashArray:"3,2",fillOpacity:0}).addTo(map).bindTooltip(`${d.name} (${d.count} tickets)`,{permanent:false,direction:"top",className:"dist-label"});
      });
      
      setTimeout(()=>map.invalidateSize(),200);
    };
    loadLeaflet();
    return()=>{if(mapInst.current){mapInst.current.remove();mapInst.current=null;markersRef.current=[];}};
  },[]);
  
  if(!coords||!coords.length) return <div style={{color:X.mu,padding:20}}>No coordinates</div>;
  return(<div>
    <div ref={mapRef} style={{width:"100%",height:420,borderRadius:8,overflow:"hidden",border:`1px solid ${X.bd}`}}/>
  </div>);
};
